User {
    int user_id PK
    varchar username
    varchar phone
    varchar password_hash
    varchar role
    varchar status
    datetime created_at
    varchar verification_code
    datetime code_expires_at
}
Channel {
    int channel_id PK
    int user_id FK
    varchar channel_name
    varchar status
    int subscribers_count
    int followers_count
}
Stream {
    int stream_id PK
    int channel_id FK
    varchar title
    varchar stream_key
    varchar status
    datetime created_at
    varchar video_url
}
VOD {
    int vod_id PK
    int stream_id FK
    int channel_id FK
    varchar title
    varchar status
    datetime created_at
    datetime expires_at
    varchar video_url
}
Message {
    int message_id PK
    int stream_id FK
    int user_id FK
    text content
    datetime created_at
}
Subscription {
    int subscription_id PK
    int user_id FK
    int channel_id FK
    varchar status
    datetime created_at
    datetime expires_at
}
Follower {
    int user_id PK, FK
    int channel_id PK, FK
    datetime created_at
}
Moderation {
    int moderation_id PK
    int moderator_id FK
    int target_user_id FK
    int channel_id FK
    varchar type
    datetime created_at
    datetime expires_at
}
Report {
    int report_id PK
    int reporter_id FK
    int target_user_id FK
    varchar status
    text reason
    datetime created_at
}

Channel_Moderator {
    int user_id PK, FK
    int channel_id PK, FK
    datetime created_at
}

Transaction {
    int transaction_id PK
    int user_id FK
    int channel_id FK
    decimal amount
    decimal platform_fee
    datetime created_at
}
